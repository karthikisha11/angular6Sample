SetPreviewPages
SetPreviewPages:
Default value : () => { const lastPage = this.lastPage; const currentPage = this.currentPage; if (currentPage < 4 && lastPage >= 5) { // If there are 5 or more pages and current page is less than 4... let previewPages = []; for (let i = 1; i <= 5; i++) { previewPages.push(i); } this.previewPages = previewPages; } else if (currentPage > 3 && currentPage <= lastPage - 3) { // If current page is more than 3 and less than or equal to (lastPage - 3) this.previewPages = [ currentPage - 2, currentPage - 1, currentPage, currentPage + 1, currentPage + 2 ]; } else if (currentPage > lastPage - 3 && lastPage > 4) { // If current page is greater than (lastPage - 3) and last page is more than 4... this.previewPages = [ lastPage - 4, lastPage - 3, lastPage - 2, lastPage - 1, lastPage ]; } else if (lastPage < 5) { let previewPages = []; for (let i = 1; i <= lastPage; i++) { previewPages.push(i); } this.previewPages = previewPages; } // Setting style for the page numbers... let previewPageStyles = this.previewPages.map(p => { if (p === this.currentPage) { return { "background-color": "rgba(0,0,0,0.2)", "font-weight": "bold" }; } else return { "background-color": "", "font-weight": "" }; }); this.previewPageStyles = previewPageStyles; }
